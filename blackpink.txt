    // simple tab system
    document.querySelectorAll('#tabs .tab').forEach(t=>t.addEventListener('click',()=>{
      document.querySelectorAll('#tabs .tab').forEach(x=>x.classList.remove('active'));
      t.classList.add('active');
      document.querySelectorAll('.tab-content').forEach(s=>s.style.display='none');
      document.getElementById(t.dataset.tab).style.display='block';
    }))

    // carousels
    function initCarousel(root){
      const track = root.querySelector('.carousel-track');
      const slides = track.children; let index=0;
      root.querySelectorAll('[data-action]').forEach(btn=>btn.addEventListener('click',e=>{
        if(e.target.dataset.action==='next'){ index=(index+1)%slides.length }
        else { index = (index-1+slides.length)%slides.length }
        track.style.transform = `translateX(-${index*100}%)`;
      }))
    }
    document.querySelectorAll('.carousel').forEach(initCarousel);

    // per-member carousels
    document.querySelectorAll('[data-carousel]').forEach(initCarousel);
